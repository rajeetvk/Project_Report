\begin{frame}{Slide 1: Intro and Objective}
    \textbf{Objective:} Detect emotions and flag ambiguity using TDA.
    \begin{itemize}
        \item Traditional models fail on ambiguous text.
        \item This project uses geometric features to detect uncertainty.
    \end{itemize}
\end{frame}

\begin{frame}{Slide 2: Background}
    \begin{itemize}
        \item BERT is the baseline for classification.
        \item Problem: Models are overconfident on "mixed" emotions.
        \item Solution: Topological Data Analysis (TDA).
    \end{itemize}
\end{frame}

\begin{frame}{Slide 3: Proposed Architecture}
    \begin{columns}

            \begin{enumerate}
                \item Text $\rightarrow$ BERT
                \item TDA Persistence
                \item Feature Fusion
                \item XGBoost Result
            \end{enumerate}
            \textbf{Tools:} Python, Giotto-TDA
        \end{column}
        \begin{column}{0.6\textwidth}
            \centering
