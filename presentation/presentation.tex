\begin{frame}{Slide 1: Introduction and Objective}
    \textbf{Introduction}
    \begin{itemize}
        \item Emotion detection from Reddit text is often hindered by ambiguity.
        \item Conventional models force classifications, leading to errors.
    \end{itemize}
    \vspace{0.2cm}
    \textbf{Objective}
    \begin{itemize}
        \item Use \textbf{Topological Data Analysis (TDA)} to detect geometric uncertainty.
        \item Explicitly flag "Ambiguous" inputs instead of picking a wrong class.
    \end{itemize}
    \vspace{0.2cm}
    \textbf{Contribution}
    \begin{itemize}
        \item Combined BERT, TF-IDF, and TDA persistence landscapes.
        \item Hybrid rule-based ambiguity detection layer.
    \end{itemize}
\end{frame}

\begin{frame}{Slide 2: Background and Existing Work}
    \begin{itemize}
        \item \textbf{Existing Work:} BERT/Transformer models achieve high accuracy on clear data but struggle with sarcasm or mixed emotions.
        \item \textbf{Problems:} Forced classification leads to overconfidence and misinformation.
        \item \textbf{Requirement:} A system that understands its own semantic boundaries.
        \item \textbf{TDA Solution:} Analyzes the "shape" of the local training data around the input. Loops in this shape indicate semantic overlaps (ambiguity).
    \end{itemize}
\end{frame}

\begin{frame}{Slide 3: Proposed System and Architecture}
    \begin{columns}[T]
        \begin{column}{0.4\textwidth}
            \small
            \textbf{System Steps:}
            \begin{enumerate}
                \item \textbf{Input:} Raw Text
                \item \textbf{BERT:} 768-dim vector
                \item \textbf{TDA:} Analyze geometry of 24 neighbours
                \item \textbf{Fusion:} Merge features
                \item \textbf{Output:} Label or Ambiguity
            \end{enumerate}
            \vspace{0.2cm}
            \textbf{Tools:}
            \begin{itemize}
                \item Python 3.10
                \item Giotto-TDA, XGBoost
                \item Sentence-BERT
            \end{itemize}
        \end{column}
        \begin{column}{0.6\textwidth}
