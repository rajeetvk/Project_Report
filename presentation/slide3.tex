        \begin{column}{0.55\textwidth}
            \small
            \textbf{1. Steps (Input to Output)}
            \begin{itemize}
                \item \textbf{Input:} Raw Reddit comment.
                \item \textbf{Process:} Semantic encoding + Geometric analysis.
                \item \textbf{Output:} Labelled emotion or Ambiguity flag.
            \end{itemize}

            \vspace{0.2cm}
            \textbf{2. Tools \& Technology}
            \begin{itemize}
                \item \textbf{Language:} Python 3.10+
                \item \textbf{Libraries:} Giotto-TDA, XGBoost, Transformers (Sentence-BERT).
            \end{itemize}

            \vspace{0.2cm}
            \textbf{3. Working Principle}
            \begin{enumerate}
                \item Retrieve training-set neighbours for the input text.
                \item Analyze neighborhood topology (persistent homology).
                \item Use "Loop Count" and "Confidence Gap" to flag high-uncertainty inputs.
            \end{enumerate}
